// api/track.js — Universal courier tracking // Supports: ST Courier, DTDC, Professional Courier, India Post import { initializeApp, cert, getApps } from 'firebase-admin/app'; import { getFirestore } from 'firebase-admin/firestore'; function initFirebase() { if (getApps().length === 0) { initializeApp({ credential: cert({ projectId: process.env.FIREBASE_PROJECT_ID, clientEmail: process.env.FIREBASE_CLIENT_EMAIL, privateKey: process.env.FIREBASE_PRIVATE_KEY?.replace(/\\n/g, '\n'), }), }); } return getFirestore(); } // ── Helper: fetch with browser-like headers ── async function fetchPage(url, options = {}) { return fetch(url, { headers: { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 Chrome/120.0.0.0 Safari/537.36', 'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8', 'Accept-Language': 'en-US,en;q=0.5', ...options.headers }, ...options }); } // ── Map raw status to our order status ── function mapStatus(htmlLower) { if (htmlLower.includes('delivered')) return { mapped: 'delivered', raw: 'Delivered' }; if (htmlLower.includes('out for delivery')) return { mapped: 'dispatched', raw: 'Out for Delivery' }; if (htmlLower.includes('in transit')) return { mapped: 'dispatched', raw: 'In Transit' }; if (htmlLower.includes('reached')) return { mapped: 'dispatched', raw: 'Reached Destination' }; if (htmlLower.includes('picked up') || htmlLower.includes('booked') || htmlLower.includes('manifested')) return { mapped: 'dispatched', raw: 'Picked Up' }; if (htmlLower.includes('dispatched')) return { mapped: 'dispatched', raw: 'Dispatched' }; return { mapped: null, raw: 'In Transit' }; } // ── Extract text from HTML rows ── function extractRows(html, limit = 5) { const rows = []; const rowRegex = /]*>([\s\S]*?)<\/tr>/gi; let m; while ((m = rowRegex.exec(html)) !== null) { const text = m[1].replace(/<[^>]+>/g, ' ').replace(/\s+/g, ' ').trim(); if (text.length > 8 && !/tracking\s*id|awb|consignment/i.test(text)) { rows.push(text); } } return rows.slice(0, limit).join(' → '); } // ════════════════════════════════════ // ST COURIER // ════════════════════════════════════ async function trackSTCourier(trackingId) { try { const res = await fetchPage(`https://www.stcourier.com/tracking.php?trackid=${trackingId}`); if (!res.ok) throw new Error(`HTTP ${res.status}`); const html = await res.text(); const htmlLower = html.toLowerCase(); if (htmlLower.includes('not found') || htmlLower.includes('invalid')) { return { success: false, error: 'Tracking ID not found on ST Courier' }; } const { mapped, raw } = mapStatus(htmlLower); const details = extractRows(html); return { success: true, rawStatus: raw, mappedStatus: mapped, details }; } catch (err) { return { success: false, error: 'Could not reach ST Courier: ' + err.message }; } } // ════════════════════════════════════ // DTDC // ════════════════════════════════════ async function trackDTDC(trackingId) { try { // DTDC has a JSON API endpoint const res = await fetchPage( `https://www.dtdc.in/trace/TrackingApi?action=track_single&cno=${trackingId}`, { headers: { 'Referer': 'https://www.dtdc.in/', 'Accept': 'application/json' } } ); if (!res.ok) throw new Error(`HTTP ${res.status}`); const text = await res.text(); // Try JSON parse first try { const data = JSON.parse(text); const events = data?.shipmentTrackingDetails || data?.trackingDetails || []; if (events.length > 0) { const latest = events[0]; const rawStatus = latest.status || latest.remarks || 'In Transit'; const { mapped } = mapStatus(rawStatus.toLowerCase()); const details = events.slice(0, 3) .map(e => `${e.date || ''} ${e.location || ''} ${e.status || e.remarks || ''}`.trim()) .join(' → '); return { success: true, rawStatus, mappedStatus: mapped, details }; } } catch(_) {} // Fall back to HTML scraping const htmlLower = text.toLowerCase(); if (htmlLower.includes('not found') || htmlLower.includes('invalid')) { return { success: false, error: 'Tracking ID not found on DTDC' }; } const { mapped, raw } = mapStatus(htmlLower); const details = extractRows(text); return { success: true, rawStatus: raw, mappedStatus: mapped, details }; } catch (err) { return { success: false, error: 'Could not reach DTDC: ' + err.message }; } } // ════════════════════════════════════ // PROFESSIONAL COURIER (TPC India) // ════════════════════════════════════ async function trackProfessional(trackingId) { try { // TPC India tracking page const res = await fetchPage( `https://www.tpcindia.com/tracking.php?id=${trackingId}&type=0&service=0`, { headers: { 'Referer': 'https://www.tpcindia.com/' } } ); if (!res.ok) throw new Error(`HTTP ${res.status}`); const html = await res.text(); const htmlLower = html.toLowerCase(); if (htmlLower.includes('not found') || htmlLower.includes('invalid') || htmlLower.includes('no record')) { return { success: false, error: 'Tracking ID not found on Professional Courier' }; } const { mapped, raw } = mapStatus(htmlLower); const details = extractRows(html); return { success: true, rawStatus: raw, mappedStatus: mapped, details }; } catch (err) { return { success: false, error: 'Could not reach Professional Courier: ' + err.message }; } } // ════════════════════════════════════ // INDIA POST (Official API) // ════════════════════════════════════ async function trackIndiaPost(trackingId) { try { // India Post official tracking API const res = await fetch( `https://www.indiapost.gov.in/VAS/api/Tracking/GetTrackingData?id=${trackingId}`, { headers: { 'Accept': 'application/json', 'Referer': 'https://www.indiapost.gov.in/', 'User-Agent': 'Mozilla/5.0 Chrome/120.0.0.0' } } ); if (!res.ok) throw new Error(`HTTP ${res.status}`); const data = await res.json(); // India Post API response format const events = data?.data?.trackingData || data?.trackingData || []; if (events.length > 0) { const latest = events[events.length - 1]; const rawStatus = latest.eventType || latest.description || 'In Transit'; const { mapped } = mapStatus(rawStatus.toLowerCase()); const details = events.slice(-3).reverse() .map(e => `${e.eventDate || ''} ${e.office || ''}: ${e.eventType || ''}`.trim()) .join(' → '); return { success: true, rawStatus, mappedStatus: mapped, details }; } // Fallback — try HTML page const htmlRes = await fetchPage(`https://www.indiapost.gov.in/VAS/Pages/trackconsignment.aspx`); return { success: false, error: 'Could not fetch India Post status. Please check the India Post website.' }; } catch (err) { return { success: false, error: 'Could not reach India Post: ' + err.message }; } } // ════════════════════════════════════ // MAIN HANDLER // ════════════════════════════════════ export default async function handler(req, res) { res.setHeader('Access-Control-Allow-Origin', '*'); res.setHeader('Access-Control-Allow-Methods', 'POST, OPTIONS'); res.setHeader('Access-Control-Allow-Headers', 'Content-Type'); if (req.method === 'OPTIONS') return res.status(200).end(); if (req.method !== 'POST') return res.status(405).json({ error: 'Method not allowed' }); const { orderId, trackingId, courier } = req.body; if (!orderId || !trackingId || !courier) { return res.status(400).json({ error: 'orderId, trackingId and courier required' }); } // Route to correct courier tracker let result; switch (courier) { case 'ST Courier': result = await trackSTCourier(trackingId); break; case 'DTDC': result = await trackDTDC(trackingId); break; case 'Professional Courier': result = await trackProfessional(trackingId); break; case 'India Post': result = await trackIndiaPost(trackingId); break; default: return res.status(200).json({ success: false, error: `Auto-tracking not available for ${courier}. Please visit their website.` }); } if (!result.success) { return res.status(200).json(result); } // Update Firestore try { const db = initFirebase(); const updates = { 'tracking.rawStatus': result.rawStatus, 'tracking.details': result.details || '', 'tracking.fetchedAt': new Date().toISOString(), updatedAt: new Date().toISOString() }; if (result.mappedStatus === 'delivered') { updates.status = 'delivered'; } await db.collection('orders').doc(orderId).update(updates); } catch (err) { console.error('Firestore update error:', err); } return res.status(200).json({ success: true, rawStatus: result.rawStatus, mappedStatus: result.mappedStatus, details: result.details || '', fetchedAt: new Date().toISOString() }); }
